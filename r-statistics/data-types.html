List¶

  • The components used to form the list are copied when forming the new list and the originals are not affected.

lst <- list(name="Fred", wife="Mary", no.children=3, child.ages=c(4,7,9))

lst
$name

$wife
'Mary'
$no.children

$child.ages

  2. 7
  3. 9
lst$name
'Fred'
lst[[4]][1]
4
lst[["no.children"]]
3
n = "child.ages"
lst[[n]]
  1. 4
  2. 7
  3. 9
lst[1]
$name = 'Fred'
lst[1] = "Freddy"
lst[1]
$name = 'Freddy'
lst[[1]]
'Freddy'
lst[["name"]]
'Freddy'
lst[[1]] = "Froggy"
lst[1]
$name = 'Froggy'
lst[[1]]
'Froggy'

Names of each list component is abbreviated to the minimum characters needed.

lst$na
'Froggy'
lst$no
3

dframe <- data.frame (
  Training = c("Strength", "Stamina", "Other"),
  Pulse = c(100, 150, 120),
  Duration = c(60, 30, 45)
)

# Add a new row
new_dframe <- rbind(dframe, list(Training="Force", Duration=110, Pulse=100))

Factor¶

music = c("Jazz", "Rock", "Classic", "Classic", "Pop", "Jazz", "Rock", "Jazz")
music_factors = factor(music)
music_factors
  1. Jazz
  2. Rock
  3. Classic
  4. Classic
  5. Pop
  6. Jazz
  7. Rock
  8. Jazz
Levels:
  1. 'Classic'
  2. 'Jazz'
  3. 'Pop'
  4. 'Rock'
lvls = levels(music_factors)
lvls
  1. 'Classic'
  2. 'Jazz'
  3. 'Pop'
  4. 'Rock'
nlevels(music_factors)
4
